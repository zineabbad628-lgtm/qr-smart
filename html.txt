<script>
    // 1. تحديد الروابط الثلاثة الخاصة بك هنا
    const pages = {
        page1: "https://yourdomain.com/page1", // استبدله برابط صفحتك الأولى
        page2: "https://yourdomain.com/page2", // استبدله برابط صفحتك الثانية
        page3: "https://yourdomain.com/page3"  // استبدله برابط صفحتك الثالثة
    };

    // 2. قراءة المعاملات (Parameters) من الرابط
    const urlParams = new URLSearchParams(window.location.search);
    const target = urlParams.get('to'); // يبحث عن كلمة ?to= في الرابط

    // 3. التوجيه التلقائي الذكي
    if (target && pages[target]) {
        // تغيير واجهة الصفحة لإعلام المستخدم أنه جاري تحويله
        document.getElementById('app').innerHTML = `
            <div class="loading">جاري تحويلك تلقائياً...</div>
            <p>إذا لم يتم تحويلك خلال ثوانٍ، <a href="${pages[target]}">اضغط هنا</a>.</p>
        `;
        
        // الانتقال الفوري للصفحة المطلوبة
        window.location.href = pages[target];
    }
</script>
